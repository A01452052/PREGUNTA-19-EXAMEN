# === FUNCIN PARA PRUEBAS DE ESTACIONARIEDAD ===
def test_stationarity(series, name, confidence=0.05):
    print(f"\n{'='*60}")
    print(f"PRUEBA DE ESTACIONARIEDAD: {name}")
    print(f"{'='*60}")

    # --- ADF Test ---
    adf = adfuller(series, autolag='AIC')
    print(f"ADF Test (H0: tiene raz unitaria)")
    print(f"   Estadstico: {adf[0]:.6f}")
    print(f"   p-value: {adf[1]:.6f}")
    print(f"   Crticos: 1%={adf[4]['1%']:.3f}, 5%={adf[4]['5%']:.3f}, 10%={adf[4]['10%']:.3f}")
    adf_decision = "ESTACIONARIA" if adf[1] < confidence else "NO ESTACIONARIA"
    print(f"    ADF: {adf_decision} (p < 0.05)")

    # --- KPSS Test ---
    kpss_stat, p_value, lags, crit = kpss(series, regression='c', nlags="auto")
    print(f"\nKPSS Test (H0: es estacionaria)")
    print(f"   Estadstico: {kpss_stat:.6f}")
    print(f"   p-value: {p_value:.6f}")
    print(f"   Crticos: 1%={crit['1%']:.3f}, 5%={crit['5%']:.3f}, 10%={crit['10%']:.3f}")
    kpss_decision = "ESTACIONARIA" if p_value > confidence else "NO ESTACIONARIA"
    print(f"    KPSS: {kpss_decision} (p > 0.05)")

    # --- Conclusin combinada ---
    print(f"\nConclusin combinada:")
    if adf[1] < 0.05 and p_value > 0.05:
        print("   Serie ESTACIONARIA (ambas pruebas coinciden)")
    elif adf[1] > 0.05 and p_value < 0.05:
        print("   Serie NO ESTACIONARIA (raz unitaria o tendencia)")
    else:
        print("   Resultados mixtos  revisar diferenciacin")

    return adf, (kpss_stat, p_value, lags, crit)
